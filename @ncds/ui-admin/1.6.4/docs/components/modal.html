Modal

Modal은 사용자에게 중요한 정보를 표시하거나 작업을 수행하기 위한 오버레이 창입니다. 모달은 현재 작업을 중단하고 사용자의 주의를 필요로 하는 내용에 집중시킵니다.

인터랙티브 예제​

아래 인터랙티브 예제를 통해 다양한 모달 속성을 실시간으로 변경하고 동작을 확인해볼 수 있습니다.

제목은 한문장으로 입력해주세요.

모달 내용을 이곳에 작성합니다.

기본 사용법​

import { Modal, Button } from '@ncds/ui-admin';

function App() {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <>
      <button onClick={() => setIsOpen(true)}>모달 열기</button>
      <Modal isOpen={isOpen} onClose={() => setIsOpen(false)}>
        <Modal.Header
          title="제목은 한문장으로 입력해주세요."
          subtitle="최대 글자수 공백 포함 70자까지 입력 가능합니다."
          onClose={() => setIsOpen(false)}
        />
        <Modal.Content>모달 내용을 이곳에 작성합니다.</Modal.Content>
      </Modal>
    </>
  );
}

HTML 사용법​

HTML에서 직접 사용할 수 있습니다. ncua.Modal 클래스를 사용하여 모달을 생성하고 관리합니다.

기본 사용법​

<!DOCTYPE html>
<html lang="ko">
  <head>
    <meta charset="UTF-8" />
    <title>Modal Example</title>
    <link rel="stylesheet" href="path/to/ui-admin.css" />
  </head>
  <body>
    <button id="openModal">모달 열기</button>

    <script src="path/to/ui-admin.bundle.js"></script>
    <script>
      // 1. 기본 모달 생성
      const modal = new ncua.Modal({
        size: 'md',
        closeOnBackdropClick: true,
        closeOnEsc: true,
        zIndex: 2000,
        onClose: () => {
          console.log('모달이 닫혔습니다');
        },
      });

      // 2. 모달 헤더 생성
      const header = new ncua.Modal.Header({
        title: '모달 제목',
        subtitle: '모달 부제목',
      });
      header.setCloseHandler(() => modal.close());

      // 3. 모달 콘텐츠 생성
      const content = new ncua.Modal.Content('<p>모달 내용입니다.</p>');

      // 4. 모달 액션 버튼 생성
      const actions = new ncua.Modal.Actions('', {
        layout: 'horizontal',
        align: 'stretch',
      });

      // 버튼 생성
      const cancelBtn = document.createElement('button');
      cancelBtn.className = 'ncua-btn ncua-btn--md ncua-btn--secondary-gray';
      cancelBtn.innerHTML = '<span class="ncua-btn__label">취소</span>';
      cancelBtn.addEventListener('click', () => modal.close());

      const confirmBtn = document.createElement('button');
      confirmBtn.className = 'ncua-btn ncua-btn--md ncua-btn--primary';
      confirmBtn.innerHTML = '<span class="ncua-btn__label">확인</span>';
      confirmBtn.addEventListener('click', () => {
        console.log('확인 버튼 클릭');
        modal.close();
      });

      actions.setContent([cancelBtn, confirmBtn]);

      // 5. 모달 구성 요소 조립
      const modalEl = modal.getModalElement();
      header.appendTo(modalEl);
      content.appendTo(modalEl);
      actions.appendTo(modalEl);

      // 6. 모달 열기
      document.getElementById('openModal').addEventListener('click', () => {
        modal.open();
      });
    </script>
  </body>
</html>
